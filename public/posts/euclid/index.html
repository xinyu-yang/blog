#include <iostream>                    
using namespace std;                   
                                       
void extEuc(int a, int b, int&x, int&y)
{                                      
    int rx,ry;                         
    int r(0);                          
    if (a%b==0)                        
    {                                  
        x=0;y=1;                       
        return;                        
    }                                  
    else                               
    {                                  
        r = a%b;                       
        extEuc(b, r, rx, ry);          
        x = ry;                        
        y = rx - ry*a/b;               
        return;                        
    }                                  
}                                      
                                       
int main()                             
{                                      
    int a,b;                           
    int x,y;                           
    cin >> a >> b;                     
    extEuc(a, b, x, y);                
    cout << x <<' '<< y << endl;       
    return 0;                          
}

int main()                      
{                               
    int a,b;
